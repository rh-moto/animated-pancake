Sub EndPlusOne()
    Dim i As Long, F, RangeList As String, LastCell As Range, Farray, Sarray As Variant
    
    ' Working on the chart currently selected
    With Application.ActiveChart
        
        ' If no chart is selected, raise error message
        On Error GoTo Err_Msg
        
        ' List of ranges updated (need to store to avoid updating the label range  multiple times)
        RangeList = ""
        
        ' Run loop for all series in the active chart
        For i = 1 To .SeriesCollection.Count
        
            ' Extract formula for each series, and split it by "'"
            F = .SeriesCollection(i).Formula
            Farray = Split(F, ",")
            
            ' adjust the data range
            For j = 1 To UBound(Farray)
                ' Split range by $
                Sarray = Split(Farray(j), "$")
                
                ' If the length of the array is shorter than 4 (namely, the series name cell), just continue
                If UBound(Sarray) < 4 Then
                    GoTo continue
                
                ' Forgot why this elseif is here
                'ElseIf Sarray(2) = Sarray(4) Then
                '    GoTo continue
                    
                Else
                    ' If the target array in the formula is already inside the RangeList (namely it's already updated),  just continue
                    If InStr(RangeList, Farray(j)) > 0 Then
                        GoTo continue
                    
                    ' update the range and add it into the RangeList
                    Else
                        Sarray(4) = Sarray(4) + 1
                        Farray(j) = Join(Sarray, "$")
                        RangeList = RangeList & ";" & Farray(j)
                    End If
                    
                End If
                
continue:
            Next j
            
            F = Join(Farray, ",")
            .SeriesCollection(i).Formula = F
            'MsgBox F
        
        Next i
    
    End With
    
    Exit Sub
    
Err_Msg:
    MsgBox "Please select a chart to use this shortcut"

End Sub
